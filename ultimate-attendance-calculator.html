Ultimate Attendance Calculator (India)

Calculate your attendance %, required classes, safe bunk days & predicted percentage.

<script>
(function(){
  function clamp(v,min,max){
    return Math.max(min, Math.min(max, v));
  }

  function attendanceCalcRun(){
    var totalEl = document.getElementById('att_total');
    if(!totalEl){ return; } // not on this page

    var total       = Number(document.getElementById('att_total').value || 0);
    var attended    = Number(document.getElementById('att_attended').value || 0);
    var requiredPct = Number(document.getElementById('att_required').value || 0);
    var future      = Number(document.getElementById('att_future').value || 0);
    var plannedAbs  = Number(document.getElementById('att_plannedAbs').value || 0);
    var precision   = Number(document.getElementById('att_precision').value || 0);

    if(total <= 0){
      alert('Total classes must be greater than 0.');
      return;
    }
    if(attended < 0){
      alert('Attended classes cannot be negative.');
      return;
    }

    var r = clamp(requiredPct, 0, 100) / 100;

    // Current %
    var curPct = (attended / total) * 100;

    // Classes needed to reach target if you attend all from now
    // (attended + x) / (total + x) >= r  -> x >= (r*total - attended)/(1-r)
    var classesNeeded = null;
    if(attended / total >= r){
      classesNeeded = 0;
    } else {
      var denom = (1 - r);
      if(denom <= 0){
        classesNeeded = null;
      } else {
        classesNeeded = Math.ceil((r*total - attended) / denom);
        if(classesNeeded < 0) classesNeeded = 0;
      }
    }

    // Classes you can miss now (current bunk capacity)
    // attended / (total + m) >= r  -> m <= (attended - r*total) / r
    var canMiss = 0;
    var numerator = (attended - r*total);
    if(numerator <= 0){
      canMiss = 0;
    } else {
      canMiss = Math.floor(numerator / r);
      if(canMiss < 0) canMiss = 0;
    }

    // Predicted attendance after future classes
    var futureAttend = Math.max(0, future - plannedAbs);
    var predictedPct = ((attended + futureAttend) / (total + future)) * 100;

    // Future miss allowance:
    // (attended + (future - m)) / (total + future) >= r
    // => m <= attended + future - r*(total + future)
    var futureMiss = Math.floor(attended + future - r*(total + future));
    if(futureMiss < 0) futureMiss = 0;
    if(futureMiss > future) futureMiss = future; // can't miss more than total future classes

    function fmt(v){
      return Number.isFinite(v) ? v.toFixed(precision) : '—';
    }

    var curEl = document.getElementById('att_curPercent');
    var needEl = document.getElementById('att_classesNeeded');
    var canMissEl = document.getElementById('att_canMiss');
    var predEl = document.getElementById('att_predicted');
    var futMissEl = document.getElementById('att_futureMiss');
    var statusEl = document.getElementById('att_statusText');

    if(curEl) curEl.textContent = fmt(curPct) + '%';
    if(needEl) needEl.textContent = (classesNeeded === null ? '—' : classesNeeded);
    if(canMissEl) canMissEl.textContent = canMiss;
    if(predEl) predEl.textContent = fmt(predictedPct) + '%';
    if(futMissEl) futMissEl.textContent = futureMiss;

    if(statusEl){
      if(curPct >= requiredPct){
        statusEl.textContent = 'On track ✅';
      } else {
        statusEl.textContent = 'Below target ⚠️';
      }
    }

    drawAttendanceChart(curPct, predictedPct);
  }

  function drawAttendanceChart(curPct, predictedPct){
    var canvas = document.getElementById('att_chart');
    if(!canvas || !canvas.getContext){ return; }

    var dpr = window.devicePixelRatio || 1;
    var rect = canvas.getBoundingClientRect();
    canvas.width = rect.width * dpr;
    canvas.height = rect.height * dpr;

    var ctx = canvas.getContext('2d');
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);

    var w = rect.width;
    var h = rect.height;

    ctx.clearRect(0,0,w,h);

    // background
    ctx.fillStyle = 'rgba(148,163,184,0.15)';
    ctx.fillRect(0,0,w,h);

    var maxBarH = h - 30;
    var barW = 40;
    var gap = 30;
    var x1 = 30;
    var x2 = x1 + barW + gap;

    // current bar
    var curH = clamp(curPct, 0, 100) / 100 * maxBarH;
    ctx.fillStyle = '#06b6d4';
    ctx.fillRect(x1, h-10-curH, barW, curH);

    // predicted bar
    var predH = clamp(predictedPct, 0, 100) / 100 * maxBarH;
    ctx.fillStyle = '#7c3aed';
    ctx.fillRect(x2, h-10-predH, barW, predH);

    ctx.fillStyle = '#e2e8f0';
    ctx.font = '11px system-ui, -apple-system, BlinkMacSystemFont, sans-serif';
    ctx.fillText('Current', x1-2, h-2);
    ctx.fillText('Predicted', x2-6, h-2);
  }

  function attendanceCalcReset(){
    var totalEl = document.getElementById('att_total');
    if(!totalEl){ return; }
    document.getElementById('att_total').value = 60;
    document.getElementById('att_attended').value = 42;
    document.getElementById('att_required').value = 75;
    document.getElementById('att_future').value = 20;
    document.getElementById('att_plannedAbs').value = 2;
    document.getElementById('att_precision').value = 1;
    attendanceCalcRun();
  }

  // Expose to global so HTML onclick works
  window.attendanceCalcRun = attendanceCalcRun;
  window.attendanceCalcReset = attendanceCalcReset;

  // Auto-init on page load
  document.addEventListener('DOMContentLoaded', function(){
    // only run if calculator exists on this page
    if(document.getElementById('att_total')){
      attendanceCalcRun();
      window.addEventListener('resize', function(){
        attendanceCalcRun();
      });
    }
  });
})();
</script>
